GroupB開発規約

1. ファイルの構造

ecsite.com
  ├── config [設定用PHP]
  ├── controllers [Slimコントローラー]
  ├── lib
  ├── migration [SQL]
  ├── models [データモデル]
  ├── public_html [公開フォルダ]
  ├── tests [テスト用PHP]
  ├── vendor [外部プラグイン]
  └── views [Twig用ビューテンプレート]

[設定用PHP]
config
  ├── bootstrap.php [ユーザの認証によってリダイレクトURLを制御する関数の定義]
  ├── config.php [Slim用設定データ]
  ├── const.php [各種ライブラリのパス定義]
  ├── database.php [MySQLの接続設定]
  └── routes.php [コントローラのrequire]

[Slimコントローラー]
controllers
  ├── content.php[コンテンツ詳細画面、コンテンツ内容表示画面のメソッド]
  ├── index.php [トップページ：コンテンツが並ぶページ、エラー画面]
  ├── purchase.php[購入画面、購入処理]
  └── user.php [ログイン画面表示、ログイン処理、ログアウト画面、新規登録画面表示、新規登録処理]

lib
  └── php
    ├── Db
    │   ├── Dao
    │   │   ├── Abstract.php [Db_Dao_Abstractクラス　DBハンドラの取得、トランザクションのメソッド]
    │   │   ├── Content.php [Db_Dao_Contentクラス　コンテンツに関するDao系メソッド]
    │   │   ├── Purchase.php [Db_Dao_Purchaseクラス　購入動作に関するDao系メソッド（購入チェックなど）]
    │   │   ├── User.php   [Db_Dao_Userクラス　ユーザーに関するDao系メソッド]
    │   │   └── sample.php [勝手に作ったテスト用ファイル]
    │   ├── HandlerFactohp   [oDb_HandlerFactoryクラス　DBハンドラ、DB接続系]
    │   └── Transaction.php [Db_Transacクラス　トランザクション系メソッド]
    ├── Factory.              [メソッド名の処理で簡単にインスタンスを生成する機能]
    ├── FactoryInjector.php
    ├── FormValidator[フォーム入力の誤り入力を判断する機能]
    │   ├── LoginFormValidator.php
    │   ├── RegisterMemberFormValidator.php
    │   └── UserFormValidator.php
    ├── PasswordUtil.php
    ├── Session.       [セッション関係]
    └── Validator.php

migration
  ├── create_content.sql
  ├── create_user.sql
  ├── create_user_purchase.sql
  └── init.sql

models
  ├── Content.php
  ├── Model.php
  ├── Purchase.php
  └── User.php

public_html
  ├── css
  ├── img
  ├── index.php
  └── js

test
  ├── browser
  │   └── user.js
  ├── data
  │   ├── DummyAbstractClass.php
  │   ├── DummyClass.php
  │   ├── DummyImplFactoryInjector.php
  │   ├── content-seed.xml
  │   └── user-seed.xml
  ├── lib
  │   └── php
  │       ├── Db
  │       │   ├── Dao
  │       │   │   ├── Db_Dao_AbstractTest.php
  │       │   │   ├── Db_Dao_ContentTest.php
  │       │   │   └── Db_Dao_UserTest.php
  │       │   ├── Db_HandlerFactoryTest.php
  │       │   └── Db_TransactionTest.php
  │       ├── FactoryTest.php
  │       ├── FormValidator
  │       │   ├── FormValidator_LoginFormValidatorTest.php
  │       │   ├── FormValidator_RegisterFormValidatorTest.php
  │       │   └── FormValidator_UserFormValidatorTest.php
  │       └── ValidatorTest.php
  └── models
    ├── ContentTest.php
    ├── ModelTest.php
    └── UserTest.php

vendor
  ├── Slim
  ├── Slim-Extras
  └── Twig

views
  ├── base.twig
  ├── content
  │   ├── detail.twig
  │   └── index.twig
  ├── error.twig
  ├── index.twig
  ├── purchase
  │   └── index.twig
  └── user
    ├── login.twig
    ├── logout.twig
    └── register.twig
